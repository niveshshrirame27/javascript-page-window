 // make sure to add a element in your html file with id attribute set to "result"...

 // also add a button to calculate the result...

 //Then add this function in your javascript file...

 function calculate() {

 // Get user input values for x and y

 let x = parseFloat(document.getElementById("inputX").value);
 let y = parseFloat(document.getElementById("inputY").value);

 // Check if the inputs are valid numbers

 if (!isNaN(x) && !isNaN(y)) {
 let z = x + y;
 document.getElementById("result").innerHTML = "the result of x + y is : " + z;
 } else {
 document.getElementById("result").innerHTML = "Please enter valid numbers for x and y.";
 }
 }

Exercise 2

                    
 // Dont forgrt to add a element in your html file with id attribute set to "time" to display timeElement...

 function updateClock() {
                
 const now = new Date();
 const timeElement = document.getElementById("time");
                
 let hours = now.getHours();
 const isPm = hours >= 12;
                
 if (hours > 12) {
 hours -= 12;
 }
                
 hours = hours.toString().padStart(2, "0");
 const minutes = now.getMinutes().toString().padStart(2, '0');
 const seconds = now.getSeconds().toString().padStart(2, '0');

 timeElement.textContent = `${hours}:${minutes}:${seconds} ${isPm ? "PM" : "AM"}`;

 }

 updateClock();

 setInterval(updateClock, 1000);
